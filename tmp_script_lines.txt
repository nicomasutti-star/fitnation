<script>
  document.addEventListener('DOMContentLoaded', function () {
    const tabMensual = document.getElementById('tab-mensual');
    const tabSemestral = document.getElementById('tab-semestral');
    const tabAnual = document.getElementById('tab-anual');
    const promoText = document.getElementById('promo-text');
    const priceBasic = document.getElementById('price-basic');
    const periodBasic = document.getElementById('period-basic');
    const totalBasic = document.getElementById('total-basic');
    const pricePrime = document.getElementById('price-prime');
    const periodPrime = document.getElementById('period-prime');
    const totalPrime = document.getElementById('total-prime');
    const permanenceBasic = document.getElementById('permanence-basic');
    const permanencePrime = document.getElementById('permanence-prime');

    const plansConfig = {
      mensual: {
        promo: '',
        basicPrice: '$27',
        basicPeriod: '/mes',
        basicTotal: '',
        primePrice: '$32',
        primePeriod: '/mes',
        primeTotal: '',
        primePermanence: '12 meses de fidelidad',
        showTotal: false
      },
      semestral: {
        promo: 'Paga 5 meses, entrena 6',
        basicPrice: '$22.50',
        basicPeriod: '/mes',
        basicTotal: 'Total: $135 (paga 5, entrena 6)',
        primePrice: '$26.67',
        primePeriod: '/mes',
        primeTotal: 'Total: $160 (paga 5, entrena 6)',
        primePermanence: '12 meses de fidelidad',
        showTotal: true
      },
      anual: {
        promo: 'Paga 10 meses, entrena 12',
        basicPrice: '$22.50',
        basicPeriod: '/mes',
        basicTotal: 'Total: $270 (paga 10, entrena 12)',
        primePrice: '$26.67',
        primePeriod: '/mes',
        primeTotal: 'Total: $320 (paga 10, entrena 12)',
        primePermanence: '12 meses de fidelidad',
        showTotal: true
      }
    };

    function setActiveTab(tabType, activeTabElement) {
      [tabMensual, tabSemestral, tabAnual].forEach(tab => {
        if (tab) tab.className = 'px-6 py-2 rounded-full text-sm font-bold text-gray-500 hover:text-black transition-all';
      });
      if (activeTabElement) {
        activeTabElement.className = 'px-6 py-2 rounded-full bg-white shadow-sm text-sm font-bold text-black transition-all';
      }
      const config = plansConfig[tabType];
      if (priceBasic) priceBasic.textContent = config.basicPrice;
      if (periodBasic) periodBasic.textContent = config.basicPeriod;
      if (pricePrime) pricePrime.textContent = config.primePrice;
      if (periodPrime) periodPrime.textContent = config.primePeriod;
      if (permanencePrime) permanencePrime.textContent = config.primePermanence;
      if (config.showTotal) {
        if (totalBasic) { totalBasic.textContent = config.basicTotal; totalBasic.classList.remove('hidden'); }
        if (totalPrime) { totalPrime.textContent = config.primeTotal; totalPrime.classList.remove('hidden'); }
      } else {
        if (totalBasic) totalBasic.classList.add('hidden');
        if (totalPrime) totalPrime.classList.add('hidden');
      }
      if (promoText) {
        if (config.promo) {
          promoText.textContent = config.promo;
          promoText.classList.remove('opacity-0');
        } else {
          promoText.classList.add('opacity-0');
        }
      }
    }

    if (tabMensual) tabMensual.addEventListener('click', () => setActiveTab('mensual', tabMensual));
    if (tabSemestral) tabSemestral.addEventListener('click', () => setActiveTab('semestral', tabSemestral));
    if (tabAnual) tabAnual.addEventListener('click', () => setActiveTab('anual', tabAnual));

    setActiveTab('mensual', tabMensual);
  });
</script>
